#include <math.h>
#include <stdio.h>

#define n 4

int queen[n + 1];

void Show() { /* 输出所有皇后摆放方案 */
    int i;
    printf("(");
    for (i = 1; i <= n; i++) {
        printf(" %d", queen[i]);
    }
    printf(")\n");
}

        }

}
